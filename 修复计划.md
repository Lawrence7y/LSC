```python
"""平台品牌图标 SVG path 定义与渲染。"""
from __future__ import annotations
from PySide6.QtCore import QByteArray, Qt
from PySide6.QtGui import QIcon, QPainter, QPixmap
from PySide6.QtSvg import QSvgRenderer

# 从 ui-prototype.html:427 迁移，统一 viewBox 24x24，fill 由调用方注入
_PLATFORM_SVG_PATHS: dict[str, str] = {
    "douyin": "M16.6 5.82s.51.5 0 0A4.28 4.28 0 0 1 15.54 3h-3.09v12.4...",
    "bilibili": "M17.813 4.653h.854c1.51.054 2.769.578 3.773 1.574...",
    "huya": "M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10...",
    "custom": "M3.9 12c0-1.71 1.39-3.1 3.1-3.1h4V7H7c-2.76 0-5 2.24-5 5...",
}

def render_platform_icon(key: str, color: str = "#ffffff", size: int = 14) -> QPixmap:
    """渲染平台图标为单色 QPixmap，供 _PlatformTag 使用。"""
    path = _PLATFORM_SVG_PATHS.get(key, _PLATFORM_SVG_PATHS["custom"])
    svg = (f'<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" '
           f'fill="{color}"><path d="{path}"/></svg>')
    renderer = QSvgRenderer(QByteArray(svg.encode("utf-8")))
    if not renderer.isValid():
        return QPixmap()
    pm = QPixmap(size, size)
    pm.fill(Qt.transparent)
    p = QPainter(pm)
    renderer.render(p)
    p.end()
    return pm
```

```python
def _on_mark_in(self):
    room = self._manager.get_room(self._selected_room_id) if self._selected_room_id else None

    pos = self._manager.get_preview_position(room.room_id)
```

```python
class MultiRoomManager(QObject):
    def __init__(self, ...):
        ...
        self._preview_loop_timer: QTimer | None = None

    def start_range_loop(self, room_id, start, end):
        """循环播放 [start, end]，每 200ms 检查位置，越界则 seek 回 start。"""
        self.stop_range_loop()
        self._loop_room = room_id
        self._loop_start, self._loop_end = start, end
        self.seek_preview(room_id, start)
        self._preview_loop_timer = QTimer(self)
        self._preview_loop_timer.setInterval(200)
        self._preview_loop_timer.timeout.connect(self._on_loop_tick)
        self._preview_loop_timer.start()

    def stop_range_loop(self):
        if self._preview_loop_timer:
            self._preview_loop_timer.stop()
            self._preview_loop_timer = None

    def _on_loop_tick(self):
        pos = self.get_preview_position(self._loop_room)
        if pos >= self._loop_end or pos < self._loop_start:
            self.seek_preview(self._loop_room, self._loop_start)
```

### 2.6 错误信息友好化层

**新增** `lsc/utils/error_messages.py`：

```python
"""把底层错误翻译成用户可读的中文提示。"""
import re

_PATTERNS: list[tuple[re.Pattern, str, str]] = [
    (re.compile(r"403|Forbidden", re.I),
     "access_denied",
     "平台拒绝了连接（403）。可能主播未开播，或需要登录 Cookie。"),
    (re.compile(r"Connection refused|ECONNREFUSED", re.I),
     "conn_refused",
     "无法连接到直播服务器。请检查网络或稍后重试。"),
    (re.compile(r"No space left|ENOSPC", re.I),
     "disk_full",
     "磁盘空间不足，无法继续录制。请清理输出目录。"),
    (re.compile(r"Stream not found|No stream", re.I),
     "no_stream",
     "未找到直播流。主播可能已下播。"),
    # FFmpeg 通用
    (re.compile(r"ffmpeg.*error", re.I),
     "ffmpeg_error",
     "录制引擎出错。请检查编码器设置或重启应用。"),
]

def humanize_error(raw: str) -> str:
    """返回用户友好的错误描述。未匹配则返回原文（加前缀）。"""
    if not raw:
        return ""
    for pattern, _code, msg in _PATTERNS:
        if pattern.search(raw):
            return msg
    return f"发生错误：{raw}"
```

### 3.4 撤销/重做基础框架

**问题**：删除房间、标记入/出点等操作不可撤销。这是框架级改动。

**设计**：Command 模式，仅覆盖**有破坏性的操作**（删除房间、清空切片），不追求全覆盖。

**新增** `lsc/gui/undo.py`：

```python
from dataclasses import dataclass
from typing import Callable

@dataclass
class Command:
    description: str
    undo: Callable
    redo: Callable

class UndoStack:
    def __init__(self, limit: int = 50):
        self._undo: list[Command] = []
        self._redo: list[Command] = []
        self._limit = limit

    def execute(self, cmd: Command):
        cmd.redo()
        self._undo.append(cmd)
        self._redo.clear()
        if len(self._undo) > self._limit:
            self._undo.pop(0)

    def undo(self):
        if not self._undo: return False
        cmd = self._undo.pop()
        cmd.undo()
        self._redo.append(cmd)
        return True

    def redo(self):
        if not self._redo: return False
        cmd = self._redo.pop()
        cmd.redo()
        self._undo.append(cmd)
        return True
```

**集成**：`MultiRoomPage` 持有 `self._undo = UndoStack()`，删除房间时压栈：

```python
def _on_remove(self, room_id):
    room = self._manager.get_room(room_id)
    # 保存可重建状态
    cmd = Command(
        description=f"删除房间 {room.streamer_name or room.room_url}",
        undo=lambda: self._restore_room(room),
        redo=lambda: self._do_remove(room_id),
    )
    self._undo.execute(cmd)
```

绑定 `Ctrl+Z` / `Ctrl+Y` 快捷键（`keyPressEvent`）。

**⚠️ 范围控制**：仅做删除房间、清空切片两类命令，标记入/出点不做（频繁，撤销栈会爆）。

### 5.1 线程安全：子线程写 room state

**问题**：`_BatchRecordWorker`（`manager.py:58`）在子线程调 `start_recording`，写 `room.is_recording` 等属性，与主线程 `watchdog_check` 竞态。

**改动**：worker 只做**纯计算/IO**，状态写回通过**信号回主线程**：

```python
class _BatchRecordWorker(QThread):
    # 新增信号：把要写的状态发回主线程
    room_state_ready = Signal(str, dict)  # room_id, {is_recording, output_path, error}

    def run(self):
        for room_id in self._room_ids:
            # 在子线程只做 FFmpeg 启动（IO 密集）
            result = self._manager._start_recording_io(room_id, ...)
            # 不直接写 room.*，发信号
            self.room_state_ready.emit(room_id, result)
            self.room_started.emit(room_id, result["ok"])
```

`MultiRoomManager` 在主线程槽里写 `room.is_recording = result["is_recording"]`。
